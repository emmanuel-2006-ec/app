async function apiCall(endpoint, options = {}) {
  const token = sessionStorage.getItem('token');
  const headers = {
    'Content-Type': 'application/json',
    ...(token ? { 'Authorization': `Bearer ${token}` } : {})
  };
  const res = await fetch(`${API_BASE}${endpoint}`, { ...options, headers });
  if (!res.ok) {
    let errMsg = 'Request failed';
    try {
      const errData = await res.json();
      errMsg = errData.error || errMsg;
    } catch(_) {}
    throw new Error(errMsg);
  }
  const text = await res.text();
  return text ? JSON.parse(text) : null;
}

// ===== HELPERS =====
function timeAgo(date) {
  const diff = Date.now() - new Date(date).getTime();
  const mins = Math.floor(diff / 60000);
  if (mins < 1) return 'Just now';
  if (mins < 60) return mins + 'm ago';
  const hrs = Math.floor(mins / 60);
  if (hrs < 24) return hrs + 'h ago';
  const days = Math.floor(hrs / 24);
  if (days < 7) return days + 'd ago';
  return new Date(date).toLocaleDateString();
}

}

async function initDashboard() {
  const token = sessionStorage.getItem('token');
  const userData = sessionStorage.getItem('user');
  if (!token || !userData) {
    window.location.href = '/index.html';
    return;
  }
  try {
    await apiCall('/api/users');
    currentUser = JSON.parse(userData);
    document.getElementById('dashAvatar').src = currentUser.profilePic || getDefaultAvatar();
    document.getElementById('postUserAvatar').src = currentUser.profilePic || getDefaultAvatar();
    socket = io(API_BASE || window.location.origin, { transports: ['websocket', 'polling'] });
    socket.emit('join', currentUser._id);
    // Socket listeners
    socket.on('new-notification', (data) => {
      playSound();
      renderNotifications();
      if ('Notification' in window && Notification.permission === 'granted') {
        new Notification('Mzuzu Hub', { body: data.message, icon: '/favicon.ico' });
      }
    });
    socket.on('incoming-call', (data) => {
      const { from, offer } = data;
      showIncomingCall(from, offer);
    });
    socket.on('call-answered', async (data) => {
      const { from, answer } = data;
      if (callActive && peerConnection) {
        await peerConnection.setRemoteDescription(new RTCSessionDescription(answer));
      }
    });
    socket.on('ice-candidate', async (data) => {
      const { from, candidate } = data;
      if (peerConnection) {
        try { await peerConnection.addIceCandidate(new RTCIceCandidate(candidate)); } catch (e) {}
      }
    });
    socket.on('call-ended', () => { endCall(); });
    // Discussion invite (overlay)
    socket.on('discussion-invite', (data) => {
      const { from, fromUsername, roomId } = data;
      showDiscussionIncomingCall(from, fromUsername, roomId);
    });
    // Discussion signaling
    socket.on('discussion-user-joined', (data) => {
      const { userId } = data;
      if (userId !== currentUser._id && !discussionPeers[userId]) {
        createDiscussionPeer(userId);
      }
    });
    socket.on('discussion-offer', async (data) => {
      const { from, offer } = data;
      if (!discussionPeers[from]) { createDiscussionPeer(from); }
      const pc = discussionPeers[from];
      await pc.setRemoteDescription(new RTCSessionDescription(offer));
      const answer = await pc.createAnswer();
      await pc.setLocalDescription(answer);
      socket.emit('discussion-answer', { roomId: discussionRoomId, target: from, answer });
    });
    socket.on('discussion-answer', async (data) => {
      const { from, answer } = data;
      const pc = discussionPeers[from];
      if (pc) { await pc.setRemoteDescription(new RTCSessionDescription(answer)); }
    });
    socket.on('discussion-ice', async (data) => {
      const { from, candidate } = data;
      const pc = discussionPeers[from];
      if (pc) {
        try { await pc.addIceCandidate(new RTCIceCandidate(candidate)); } catch (e) {}
      }
    });
    socket.on('discussion-user-left', (data) => {
      const { userId } = data;
      removeDiscussionTile(userId);
      if (discussionPeers[userId]) {
        discussionPeers[userId].close();
        delete discussionPeers[userId];
      }
    });
    await refreshAll();
    setupEventListeners();
    setInterval(refreshAll, 30000);
    if ('Notification' in window && Notification.permission === 'default') {
      Notification.requestPermission();
    }
  } catch (error) {
    console.error('Dashboard init error:', error);
    if (error.message.includes('Unauthorized') || error.message.includes('Invalid token')) {
      sessionStorage.removeItem('token');
      sessionStorage.removeItem('user');
      window.location.href = '/index.html';
    } else {
      showError(error.message);
    }
  }
}

async function renderPosts(containerId = 'postsContainer') {
  const container = document.getElementById(containerId);
  if (!container) return;
  try {
    const posts = await apiCall('/api/posts');
    if (!posts || posts.length === 0) {
      container.innerHTML = `<div class="empty-state"><i class="fas fa-pen-fancy"></i><h3>No posts yet</h3><p>Be the first to share something!</p></div>`;
      return;
    }
    let html = '';
    posts.forEach(post => {
      if (post.type === 'video') return;
      const isLiked = post.likes && post.likes.includes(currentUser._id);
      const likeCount = post.likes ? post.likes.length : 0;
      const comments = post.comments || [];
      const author = post.author || {};
      const pname = author.username || 'Unknown';
      const pAvatar = author.profilePic || getDefaultAvatar();
      const pYear = author.yearOfStudy || '';
      const isOwner = post.author && post.author._id === currentUser._id;
      let contentHtml = '';
      if (post.type === 'sticker') contentHtml = `<span class="sticker-display">${post.content}</span>`;
      else if (post.type === 'text') contentHtml = `<p>${post.content}</p>`;
      else if (post.type === 'audio') {
        contentHtml = `
          <div class="audio-display">
            <button onclick="playAudio('${post._id}')"><i class="fas fa-play"></i></button>
            <span style="font-size:12px;font-weight:500;">🎵 Audio message</span>
            <span style="font-size:11px;color:#6a7a8a;margin-left:auto;">${post.audioDuration || '0:05'}</span>
          </div>
        `;
      }
      if (post.imageUrl) {
        contentHtml += `<img src="${post.imageUrl}" class="post-image" alt="image" />`;
      }
      html += `
        <div class="post-card" data-postid="${post._id}">
          ${isOwner ? `<button class="delete-btn" onclick="deletePost('${post._id}')"><i class="fas fa-trash-alt"></i></button>` : ''}
          <div class="post-header">
            <img src="${pAvatar}" alt="" />
            <div class="post-user">
              <div class="pname">${pname} <span style="font-size:10px;color:#6a7a8a;">${getYearEmoji(pYear)}</span></div>
              <div class="ptime">${timeAgo(post.createdAt)}</div>
            </div>
          </div>
          <div class="post-content">${contentHtml}</div>
          <div class="post-actions">
            <button class="${isLiked ? 'liked' : ''}" onclick="toggleLike('${post._id}')">
              <i class="fas fa-heart"></i> ${likeCount > 0 ? likeCount : ''} Like
            </button>
            <button onclick="toggleComments('${post._id}')"><i class="fas fa-comment"></i> ${comments.length} Comment${comments.length !== 1 ? 's' : ''}</button>
          </div>
          <div class="post-comments" id="comments-${post._id}" style="display:${comments.length > 0 ? 'block' : 'none'};">
            ${comments.map(c => `
              <div class="comment-item">
                <img src="${c.profilePic || getDefaultAvatar()}" alt="" />
                <div class="cbody"><strong>${c.username || 'Unknown'}</strong> ${c.text}</div>
              </div>
            `).join('')}
            <div class="comment-input">
              <input type="text" placeholder="Write a comment..." id="commentInput-${post._id}" />
              <button onclick="addComment('${post._id}')">Post</button>
            </div>
          </div>
        </div>
      `;
    });
    container.innerHTML = html;
  } catch (e) {
    container.innerHTML = `<div class="empty-state"><i class="fas fa-exclamation-triangle"></i><h3>Error loading posts</h3><p>${e.message}</p></div>`;
  }
}

async function renderChatMessages(userId) {
  const container = document.getElementById('chatMessages');
  try {
    const msgs = await apiCall(`/api/messages/${userId}`);
    if (!msgs || msgs.length === 0) {
      container.innerHTML = `<div style="text-align:center;padding:16px;color:#6a7a8a;">No messages yet. Say hello! 👋</div>`;
      return;
    }
    let html = '';
    msgs.forEach(m => {
      const isSent = m.from === currentUser._id;
      let contentHtml = '';
      if (m.type === 'text') {
        contentHtml = m.content;
      } else if (m.type === 'image') {
        contentHtml = `<img src="${m.data}" class="attachment-img" alt="image" />`;
      } else if (m.type === 'audio') {
        contentHtml = `<audio controls class="attachment-audio"><source src="${m.data}" type="audio/webm" /></audio>`;
      } else if (m.type === 'document') {
        contentHtml = `<a href="${m.data}" target="_blank" class="attachment">📄 Document</a>`;
      } else {
        contentHtml = m.content || 'Attachment';
      }
      html += `
        <div class="msg ${isSent ? 'sent' : 'received'}">
          ${contentHtml}
          <span class="msg-time">${timeAgo(m.createdAt)}</span>
        </div>
      `;
    });
    container.innerHTML = html;
    container.scrollTop = container.scrollHeight;
  } catch (e) { container.innerHTML = `<div style="text-align:center;padding:16px;color:#6a7a8a;">Error loading messages</div>`; }
}

  if (this.files && this.files[0]) {
    const file = this.files[0];
    const reader = new FileReader();
    reader.onload = async function(e) {
      const dataUrl = e.target.result;
      let type = 'document';
      if (file.type.startsWith('image/')) type = 'image';
      else if (file.type.startsWith('audio/')) type = 'audio';
      await sendChatMessage(file.name, type, dataUrl);

    this.value = '';
  }

document.getElementById('voiceRecordBtn').addEventListener('click', async function() {
  if (isVoiceRecording) {
    if (voiceRecorder) voiceRecorder.stop();
    isVoiceRecording = false;
    this.classList.remove('recording');
    return;
  }
  try {
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
    voiceRecorder = new MediaRecorder(stream);
    voiceChunks = [];
    voiceRecorder.ondataavailable = e => voiceChunks.push(e.data);
    voiceRecorder.onstop = async () => {
      const blob = new Blob(voiceChunks, { type: 'audio/webm' });
      const reader = new FileReader();
      reader.onload = async function(e) {
        await sendChatMessage('Voice message', 'audio', e.target.result);
      };
      reader.readAsDataURL(blob);
    };
    voiceRecorder.start();
    isVoiceRecording = true;
    this.classList.add('recording');
  } catch (e) { alert('Microphone access denied.'); }
});

function playSound() {
  try {
    const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    const oscillator = audioCtx.createOscillator();
    const gainNode = audioCtx.createGain();
    oscillator.connect(gainNode);
    gainNode.connect(audioCtx.destination);
    oscillator.frequency.value = 800;
    oscillator.type = 'sine';
    gainNode.gain.value = 0.1;
    oscillator.start();
    setTimeout(() => {
      oscillator.stop();
    }, 150);
  } catch(e) { /* silent fail */ }
}

// ============================================================
const callConfiguration = {
  iceServers: [
    { urls: 'stun:stun.l.google.com:19302' },
    { urls: 'stun:stun1.l.google.com:19302' }
  ]
};

async function startCall(video = true) {
  if (!currentChatTarget) return;
  try {
    localStream = await navigator.mediaDevices.getUserMedia({
      audio: { echoCancellation: true, noiseSuppression: true },
      video: video
    });
    document.getElementById('localVideo').srcObject = localStream;
    document.getElementById('localVideo').style.display = 'block';
    peerConnection = new RTCPeerConnection(callConfiguration);
    localStream.getTracks().forEach(track => peerConnection.addTrack(track, localStream));
    peerConnection.ontrack = (event) => {
      remoteStream = event.streams[0];
      document.getElementById('remoteVideo').srcObject = remoteStream;
      document.getElementById('remoteVideo').style.display = 'block';
    };
    peerConnection.onicecandidate = (event) => {
      if (event.candidate) {
        socket.emit('ice-candidate', { to: currentChatTarget._id, candidate: event.candidate });
      }
    };
    const offer = await peerConnection.createOffer();
    await peerConnection.setLocalDescription(offer);
    socket.emit('call-user', { to: currentChatTarget._id, offer });
    callActive = true;
    document.getElementById('callOverlay').classList.add('active');
    document.getElementById('incomingCallBox').classList.remove('show');
    document.getElementById('remoteVideo').style.display = 'block';
    incomingDiscussionData = null;
  } catch (e) {
    alert('Could not start call: ' + e.message);
    console.error(e);
  }
}

  incomingCallData = { from, offer };

document.getElementById('acceptCallBtn').addEventListener('click', async function() {
  if (incomingDiscussionData) {

    document.getElementById('callOverlay').classList.remove('active');
    incomingDiscussionData = null;
    return;
  }
  if (!incomingCallData) return;
  const { from, offer } = incomingCallData;
  try {
    localStream = await navigator.mediaDevices.getUserMedia({
      audio: { echoCancellation: true, noiseSuppression: true },
      video: true
    });
    document.getElementById('localVideo').srcObject = localStream;
    document.getElementById('localVideo').style.display = 'block';
    peerConnection = new RTCPeerConnection(callConfiguration);
    localStream.getTracks().forEach(track => peerConnection.addTrack(track, localStream));
    peerConnection.ontrack = (event) => {
      remoteStream = event.streams[0];
      document.getElementById('remoteVideo').srcObject = remoteStream;
      document.getElementById('remoteVideo').style.display = 'block';
    };
    peerConnection.onicecandidate = (event) => {
      if (event.candidate) {
        socket.emit('ice-candidate', { to: from, candidate: event.candidate });
      }
    };
    await peerConnection.setRemoteDescription(new RTCSessionDescription(offer));
    const answer = await peerConnection.createAnswer();

    socket.emit('answer-call', { to: from, answer });

async function joinDiscussion(roomId) {
  discussionRoomId = roomId;
  document.getElementById('discussionSetup').style.display = 'none';
  document.getElementById('discussionRoom').style.display = 'block';
  document.getElementById('discussionGrid').innerHTML = '';
  const discussionTab = document.querySelector('.tab-btn[data-tab="discussions"]');
  if (discussionTab) { discussionTab.click(); }
  try {
    const roomData = await apiCall(`/api/discussions/${roomId}`);
    discussionLeaderId = roomData.leader;
    localDiscussionStream = await navigator.mediaDevices.getUserMedia({
      audio: { echoCancellation: true, noiseSuppression: true },
      video: true
    });
    addDiscussionTile(currentUser._id, localDiscussionStream, true, discussionLeaderId === currentUser._id);
    socket.emit('discussion-join', { roomId });
    const participants = roomData.participants.filter(id => id !== currentUser._id);
    for (const uid of participants) {
      if (!discussionPeers[uid]) { createDiscussionPeer(uid); }
    }
    isDiscussionActive = true;
    incomingDiscussionData = null;
  } catch (e) {
    alert('Error joining discussion: ' + e.message);
    endDiscussion();
  }
}
window.joinDiscussion = joinDiscussion;

function createDiscussionPeer(userId) {
  const pc = new RTCPeerConnection(callConfiguration);
  discussionPeers[userId] = pc;
  localDiscussionStream.getTracks().forEach(track => pc.addTrack(track, localDiscussionStream));
  pc.ontrack = (event) => {
    const stream = event.streams[0];
    if (!discussionStreams[userId]) {
      discussionStreams[userId] = stream;
      addDiscussionTile(userId, stream, false, userId === discussionLeaderId);
    }
  };
  pc.onicecandidate = (event) => {
    if (event.candidate) {
      socket.emit('discussion-ice', { roomId: discussionRoomId, target: userId, candidate: event.candidate });
    }
  };
  pc.createOffer().then(offer => {
    pc.setLocalDescription(offer);
    socket.emit('discussion-offer', { roomId: discussionRoomId, target: userId, offer });
  }).catch(console.error);
}

function addDiscussionTile(userId, stream, isLocal, isLeader) {
  const grid = document.getElementById('discussionGrid');
  const existing = document.getElementById(`discussion-tile-${userId}`);
  if (existing) return;
  const tile = document.createElement('div');
  tile.id = `discussion-tile-${userId}`;
  tile.className = 'discussion-tile';
  const video = document.createElement('video');
  video.autoplay = true;
  video.playsinline = true;
  if (isLocal) video.muted = true;
  video.srcObject = stream;
  video.style.width = '100%';
  video.style.height = '100%';
  video.style.objectFit = 'cover';
  tile.appendChild(video);
  const label = document.createElement('div');
  label.className = 'tile-label';
  label.textContent = isLocal ? 'You' : 'Participant';
  tile.appendChild(label);
  if (isLeader && isLocal) {
    const badge = document.createElement('div');
    badge.className = 'leader-badge';
    badge.textContent = '👑 Leader';
    tile.appendChild(badge);
  }
  grid.appendChild(tile);
}

function removeDiscussionTile(userId) {
  const tile = document.getElementById(`discussion-tile-${userId}`);
  if (tile) tile.remove();
  delete discussionStreams[userId];
}

async function renderDiscussionFriends() {
  const container = document.getElementById('discussionFriendList');
  if (!container) return;
  try {
    const friends = await apiCall('/api/friends');
    if (friends.length === 0) {
      container.innerHTML = '<p style="color:#6a7a8a;">You have no friends to invite.</p>';
      return;
    }
    let html = '';
    friends.forEach(f => {
      html += `
        <label style="display:flex;align-items:center;gap:6px;padding:4px 10px;background:#f0f4f8;border-radius:20px;cursor:pointer;">
          <input type="checkbox" class="friend-check" value="${f._id}" />
          <img src="${f.profilePic || getDefaultAvatar()}" style="width:24px;height:24px;border-radius:50%;object-fit:cover;" />
          <span>${f.username}</span>
        </label>
      `;
    });
    container.innerHTML = html;
    container.querySelectorAll('.friend-check').forEach(cb => {
      cb.addEventListener('change', function() {
        const checked = container.querySelectorAll('.friend-check:checked');
        if (checked.length > 10) {
          this.checked = false;
          alert('You can select up to 10 friends.');
        }
      });
    });
  } catch (e) {
    container.innerHTML = `<p style="color:#e74c3c;">Error loading friends.</p>`;
  }
}

document.getElementById('audioRecordBtn').addEventListener('click', async () => {
  const btn = document.getElementById('audioRecordBtn');
  if (isRecording) {
    if (mediaRecorder) mediaRecorder.stop();
    isRecording = false;
    btn.innerHTML = '<i class="fas fa-microphone"></i> Audio';
    btn.style.color = '#e74c3c';
    return;
  }
  try {
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
    mediaRecorder = new MediaRecorder(stream);
    audioChunks = [];
    mediaRecorder.ondataavailable = e => audioChunks.push(e.data);
    mediaRecorder.onstop = () => {
      const blob = new Blob(audioChunks, { type: 'audio/webm' });
      recordedAudioBlob = blob;
      recordedAudioDuration = '0:0' + Math.floor(blob.size / 10000);
      btn.innerHTML = '<i class="fas fa-check-circle"></i> Audio Ready';
      btn.style.color = '#2ecc71';
      document.getElementById('postInput').value = '🎙️ Audio message attached';
    };
    mediaRecorder.start();
    isRecording = true;
    btn.innerHTML = '<i class="fas fa-stop-circle"></i> Stop';
    btn.style.color = '#e74c3c';
  } catch (e) { alert('Microphone access denied.'); }
});

document.getElementById('submitPostBtn').addEventListener('click', async () => {
  const input = document.getElementById('postInput');
  let content = input.value.trim();
  let type = 'text';
  let stickerType = null;
  let audioUrl = null;
  let videoUrl = null;
  let imageUrl = null;

  if (selectedSticker) { type = 'sticker'; content = selectedSticker; stickerType = selectedSticker; }
  else if (recordedAudioBlob) {
    type = 'audio';
    const reader = new FileReader();
    audioUrl = await new Promise((resolve) => {
      reader.onload = e => resolve(e.target.result);
      reader.readAsDataURL(recordedAudioBlob);
    });
  } else if (uploadedVideoData) {
    type = 'video';
    videoUrl = uploadedVideoData;
  } else if (uploadedImageData) {
    type = 'text';
    imageUrl = uploadedImageData;
  }

  if (!content && type === 'text' && !imageUrl) {
    alert('Please write something or add media.');
    return;
  }

  try {
    await apiCall('/api/posts', {
      method: 'POST',
      body: JSON.stringify({ content, type, stickerType, audioUrl, videoUrl, imageUrl, audioDuration: recordedAudioDuration })
    });
    document.getElementById('postInput').value = '';
    selectedSticker = null;
    recordedAudioBlob = null;
    uploadedVideoData = null;
    uploadedImageData = null;
    document.getElementById('postImagePreview').style.display = 'none';
    document.getElementById('stickerPicker').classList.remove('open');
    document.getElementById('audioRecordBtn').innerHTML = '<i class="fas fa-microphone"></i> Audio';
    document.getElementById('audioRecordBtn').style.color = '';
    document.getElementById('videoUploadBtn').innerHTML = '<i class="fas fa-film"></i> Video';
    document.getElementById('videoUploadBtn').style.color = '';
    document.getElementById('imageUploadBtn').innerHTML = '<i class="fas fa-image"></i> Image';
    document.getElementById('imageUploadBtn').style.color = '';
    document.getElementById('videoFileInput').value = '';
    document.getElementById('imageFileInput').value = '';
    await renderPosts();
    await renderVideos();
  } catch (e) { alert('Error creating post: ' + e.message); }
});
